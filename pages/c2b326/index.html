# useElementBounding 获取元素节点

import { useElementBounding } from "@vueuse/core";
<img
  ref="imgTarget"
  :src="data.photo"
  />
const {
  x: imgContainerX,
  y: imgContainerY,
  width: imgContainerWidth,
  height: imgContainerHeight,
} = useElementBounding(imgTarget);
